 33def cobra_to_xlsx(model: cobra.Model, output_path: str) -> None:
 34    """
 35    Converts a COBRA model to an Excel file with two sheets: one for reactions and one for metabolites.
 36    
 37    Parameters:
 38    model (cobra.Model): The COBRA model to be converted.
 39    output_path (str): The path where the Excel file will be saved.
 40
 41    Returns:
 42    None
 43    """
 44    # Extract reactions
 45    reaction_data = []
 46    for rxn in model.reactions:
 47        reaction_data.append({
 48            "ID": rxn.id,
 49            "Reaction": rxn.reaction,
 50            "GPR": rxn.gene_reaction_rule,
 51            "Lower bound": rxn.lower_bound,
 52            "Upper bound": rxn.upper_bound,
 53            "Objective": rxn.objective_coefficient,
 54            "Confidence Score": 0,
 55            "Subsystem": rxn.subsystem,
 56            "Description": rxn.name
 57        })
 58
 59    reactions_df = pd.DataFrame(reaction_data)
 60
 61    # Extract metabolites
 62    metabolite_data = []
 63    for met in model.metabolites:
 64        metabolite_data.append({
 65            "ID": met.id,
 66            "Formula": met.formula,
 67            "Description": met.name,
 68            "Compartment": met.compartment,
 69            "Charge": met.charge
 70        })
 71
 72    metabolites_df = pd.DataFrame(metabolite_data)
 73
 74    # Save to Excel
 75    with pd.ExcelWriter(output_path) as writer:
 76        reactions_df.to_excel(writer, sheet_name=RXN_SHEET, index=False)
 77        metabolites_df.to_excel(writer, sheet_name=MET_SHEET, index=False)
 78
 79    print(f"Model saved to {output_path}")

132def create_metabolic_graph(model: cobra.Model) -> nx.DiGraph:
133    """Creates a directed bipartite graph from a COBRA model.
134    Every metabolite and enzyme is a node, their relationships form edges.
135    Bipartite graph as there are no connections between enzymes or between metabolites.
136
137    Parameters:
138    model (cobra.Model): The COBRA model to be converted into a graph.
139    
140    Returns:
141    nx.DiGraph: A directed bipartite graph representing the model.
142    """
143    G = nx.DiGraph()
144
145    # Add metabolites as nodes
146    for metabolite in model.metabolites:
147        G.add_node(
148            metabolite.id,
149            type="metabolite",
150            name=metabolite.name,
151            formula=metabolite.formula,
152            compartment=metabolite.compartment,
153        )
154
155    for reaction in model.reactions:
156        # Add enzyme node
157        enzyme_node = f"{reaction.id}"
158        G.add_node(enzyme_node, type="enzyme", name=reaction.name)
159        
160        # Add edges from reactants to the reaction
161        for reactant in reaction.reactants:
162            G.add_edge(reactant.id, enzyme_node, type="reactant", reaction_id=reaction.id)
163        
164        # Add edges from the reaction to products
165        for product in reaction.products:
166            G.add_edge(enzyme_node, product.id, type="product", reaction_id=reaction.id)
167        
168        # Create additional edges for reversible reactions
169        if reaction.reversibility:
170            for product in reaction.products:
171                G.add_edge(product.id, enzyme_node, type="reverse_product", reaction_id=reaction.id)
172            for reactant in reaction.reactants:
173                G.add_edge(enzyme_node, reactant.id, type="reverse_reactant", reaction_id=reaction.id)
174
175    return G

177def xlsx_to_cobra(file_path: str, model_name: str) -> cobra.Model:
178    """Converts an Excel file containing metabolic data into a COBRA model.
179    The Excel file should contain two sheets: one for reactions and one for metabolites.
180    
181    Parameters:
182    file_path (str): Path to the Excel file.
183    model_name (str): Name of the COBRA model to be created.
184    
185    Returns:
186    cobra.Model: A COBRA model object created from the Excel data.
187    """
188    xls = pd.ExcelFile(file_path)
189
190    # Load pages
191    reactions_df = pd.read_excel(xls, sheet_name=RXN_SHEET)
192    metabolites_df = pd.read_excel(xls, sheet_name=MET_SHEET)
193
194    # Create empty COBRA model
195    model = cobra.Model(model_name)
196
197    # Add metabolites to the model
198    for _, row in metabolites_df.iterrows():
199        met = cobra.Metabolite(
200            id=row[MET_ID],
201            formula=row[MET_FORMULA],
202            name=row[MET_NAME],
203            compartment=row[MET_COMPARTMENT] if not pd.isna(row[MET_COMPARTMENT]) else DEFAULT_COMPARTMENT,
204            charge=row[MET_CHARGE],
205        )
206        model.add_metabolites([met])
207
208    # Add reactions to the model
209    for _, row in reactions_df.iterrows():
210        reaction = cobra.Reaction(id=row[RXN_ID])
211        reaction.name = row[RXN_NAME]
212        reaction.lower_bound = row[RXN_LOWER_BOUND] if not pd.isna(row[RXN_LOWER_BOUND]) else DEFAULT_LOWER_BOUND
213        reaction.upper_bound = row[RXN_UPPER_BOUND] if not pd.isna(row[RXN_UPPER_BOUND]) else DEFAULT_UPPER_BOUND
214        if not pd.isna(row[RXN_GPR]):
215            reaction.gene_reaction_rule=row[RXN_GPR]
216        reaction.subsystem=row[RXN_SUBSYSTEM]
217
218        model.add_reactions([reaction])
219        model.reactions.get_by_id(row[RXN_ID]).build_reaction_from_string(row[RXN_EQUATION]) # This function has to have access to model - therefore the reaction needs to be already present in model
220        model.reactions.get_by_id(row[RXN_ID]).objective_coefficient=row[RXN_OBJECTIVE] if not pd.isna(row[RXN_OBJECTIVE]) else 0 # Same here
221
222    return(model)

276def find_paths(G: nx.Graph, source: str, target: str, nodes: list, count=5, print_path=False) -> tuple:
277    """Finds paths in a directed graph from source to target, including all nodes.
278
279    Parameters:
280    G (nx.Graph): The graph.
281    source (str): The starting node.
282    target (str): The target node.
283    nodes (list): List of nodes that must be included in the path.
284    count (int): Number of paths to find.
285    print_path (bool): If True, prints the paths.
286
287    Returns:
288    tuple: Tuple of paths found. First one uses IDs, second one uses names.
289    """
290    paths = []
291    named_paths = []
292
293    for path in nx.shortest_simple_paths(G, source, target):
294        skip = False
295        for node in nodes:
296            if node not in path:
297                skip = True
298                break
299
300        if skip:
301            continue
302
303        count -= 1
304        paths.append(path)
305        named_path = [get_name_by_id(G, node_id) for node_id in path]
306        named_paths.append(named_path)
307        
308        if print_path:
309            print(named_path)
310
311        if count == 0:
312            break
313    
314    return paths, named_paths

321def prune_graph(G: nx.Graph, threshold: int, to_keep: list) -> nx.Graph:
322    """Prunes a graph by removing nodes with degree higher than a threshold.
323    Nodes in the to_keep list will not be removed, even if they exceed the threshold.
324
325    Parameters:
326    G (nx.Graph): The graph to be pruned.
327    threshold (int): The degree threshold for pruning.
328    to_keep (list): List of nodes that should not be removed.
329    
330    Returns:
331    nx.Graph: The pruned graph.
332    """
333    G_pruned = G.copy()
334
335    high_degree_nodes = [node for node, degree in G.degree() if (degree > threshold and node not in to_keep) or "UDP" in G.nodes[node].get("name", "")]
336    to_delete_named = [get_name_by_id(G, node) for node in high_degree_nodes]
337
338    G_pruned.remove_nodes_from(high_degree_nodes)
339
340    print(f"Removed {len(high_degree_nodes)} nodes with degree greater than {threshold}.")
341    print(f"Removed the following nodes: {to_delete_named}")
342    print(f"Remaining nodes: {len(G.nodes())}")
343    return G_pruned

371def mini_model_prune(escher_map, model_path, output_name):
372    """Prunes a COBRA model based on an Escher map.
373    Deletes all reactions and metabolites not present in the Escher map and saves the pruned model in XML, JSON, and XLSX formats.
374    
375    Parameters:
376    escher_map (str): Path to the Escher map file (JSON format).
377    model_path (str): Path to the SBML model file.
378    output_name (str): Name for the output files (without extension).
379    
380    Returns:
381    None
382    """
383    model = cobra.io.read_sbml_model(model_path)
384
385    with open(escher_map, "r") as file:
386        json_data = json.load(file)
387
388    map_reactions = set(reaction_info["bigg_id"] for reaction_info in json_data[1].get("reactions", {}).values())
389
390    reactions_to_remove = [rxn.id for rxn in model.reactions if rxn.id not in map_reactions and "Biomass" not in rxn.name]
391    model.remove_reactions(reactions_to_remove)
392
393    orphan_metabolites = [met for met in model.metabolites if len(met.reactions) == 0]
394    model.remove_metabolites(orphan_metabolites)
395
396    # Ensure models directory exists
397    os.makedirs("models", exist_ok=True)
398    base = os.path.join("models", output_name)
399    # Save in xml, json and xlsx
400    cobra.io.write_sbml_model(model, base + ".xml")
401    cobra.io.save_json_model(model, base + ".json")
402    cobra_to_xlsx(model, base + ".xlsx")
403
404    print(f"Filtered model saved as {base}")
405    print(f"Removed {len(reactions_to_remove)} reactions.")
406    print(f"Removed {len(orphan_metabolites)} metabolites.")

425def extract_bigg_id(annotation, keyword):
426    """Helper function.
427    Extracts BiGG ID from the annotation of a reaction or metabolite.
428    """
429    # note that ChatGPT was used to discuss this function heavily
430    if annotation is None:
431        return None
432
433    for i in range(annotation.getNumChildren()):
434        child = annotation.getChild(i)
435        if child.getName() == "RDF":
436            for j in range(child.getNumChildren()):
437                desc = child.getChild(j)
438                if desc.getName() == "Description":
439                    for k in range(desc.getNumChildren()):
440                        bag_wrapper = desc.getChild(k)
441                        if bag_wrapper.getName() == "is":
442                            bag = bag_wrapper.getChild(0)
443                            for l in range(bag.getNumChildren()):
444                                li = bag.getChild(l)
445                                attrs = li.getAttributes()
446                                for idx in range(attrs.getLength()):
447                                    name = attrs.getName(idx)
448                                    value = attrs.getValue(idx)
449                                    uri = attrs.getURI(idx)
450                                    if name == "resource" and "rdf" in uri and keyword in value:
451                                        return value.split("/")[-1]
452    return None

454def convert_gpr_to_string(assoc: libsbml.GeneProductAssociation) -> str:
455    """Helper function.
456    Converts a gene product association to a string.
457    """
458    if assoc is None:
459        return ""
460    if assoc.getTypeCode() == libsbml.AND_ASSOCIATION:
461        return "(" + " and ".join([convert_gpr_to_string(assoc.getAssociation(i)) for i in range(assoc.getNumAssociations())]) + ")"
462    elif assoc.getTypeCode() == libsbml.OR_ASSOCIATION:
463        return "(" + " or ".join([convert_gpr_to_string(assoc.getAssociation(i)) for i in range(assoc.getNumAssociations())]) + ")"
464    elif isinstance(assoc, libsbml.GeneProductRef):
465        return assoc.getGeneProduct()
466    return ""

468def build_bigg_id_maps(sbml_path: str) -> tuple:
469    """Helper function.
470    Builds a mapping of BiGG IDs to SBML IDs for reactions and metabolites.
471    """
472    reader = libsbml.SBMLReader()
473    doc = reader.readSBML(sbml_path)
474    model = doc.getModel()
475
476    reaction_id_map = {}
477    metabolite_id_map = {}
478
479    for reaction in model.getListOfReactions():
480        bigg_id = extract_bigg_id(reaction.getAnnotation(), "bigg.reaction")
481        if bigg_id:
482            reaction_id_map[bigg_id] = reaction.getId()
483
484    for species in model.getListOfSpecies():
485        bigg_id = extract_bigg_id(species.getAnnotation(), "bigg.metabolite")
486        if bigg_id:
487            metabolite_id_map[bigg_id] = species.getId()
488
489    return reaction_id_map, metabolite_id_map

491def escher_map_sync(escher_map_path: str, model_path: str) -> None:
492    """Synchronizes an Escher map with a COBRA model by replacing metabolite and reaction names in the map with those from the model.
493    The function reads the Escher map and the COBRA model, builds a mapping of BiGG IDs to SBML IDs, and updates the Escher map accordingly.
494    The final map is saved in maps/ directory.
495    
496    Parameters:
497    escher_map_path (str): Path to the Escher map file (JSON format).
498    model_path (str): Path to the SBML model file.
499    
500    Returns:
501    None
502    """
503    def replace_markers_in_file(filename, output_filename):
504        with open(filename, "r", encoding="utf-8") as f:
505            content = f.read()
506
507        content = decode_sbml(content)
508
509        with open(output_filename, "w", encoding="utf-8") as f:
510            f.write(content)
511    
512    def replace_values(obj, replace_dict):
513        if isinstance(obj, dict):
514            return {k: replace_values(v, replace_dict) for k, v in obj.items()}
515        elif isinstance(obj, list):
516            return [replace_values(elem, replace_dict) for elem in obj]
517        elif isinstance(obj, str):
518            for key, value in replace_dict.items():
519                # It seems that Escher automatically removes these prefixes, therefore
520                # we can't add them to the map
521                if "R_" in value or "M_" in value or "G_" in value:
522                    value = value[2:]
523                obj = obj.replace(key, value)
524            return obj
525        else:
526            return obj
527    
528    # SBML changes some characters (like "-") to a standardized naming scheme,
529    # sometimes this can produce errors https://github.com/opencobra/cobrapy/issues/936
530    replace_markers_in_file(model_path, "temp.xml")
531
532    # a mapping of reaction and metabolite names is created
533    rxn_map, met_map = build_bigg_id_maps("temp.xml")
534
535    with open(escher_map_path, "r", encoding="utf-8") as f:
536        data = json.load(f)
537
538    updated_data = replace_values(data, (met_map | rxn_map))
539
540    # Ensure maps directory exists
541    os.makedirs("maps", exist_ok=True)
542    output_map_path = os.path.join("maps", f"renamed_{os.path.basename(escher_map_path)}")
543    with open(output_map_path, "w", encoding="utf-8") as f:
544        json.dump(updated_data, f, indent=2)
545    
546    os.remove("temp.xml")
547    print(f"Updated map saved as {output_map_path}")

746def filter_efms(efm_matrix: np.ndarray, rxn_list: list, model: cobra.Model) -> np.ndarray:

748
749    Parameters:

133def create_metabolic_graph(model: cobra.Model) -> nx.DiGraph:
134    """Creates a directed bipartite graph from a COBRA model.
135    Every metabolite and enzyme is a node, their relationships form edges.
136    Bipartite graph as there are no connections between enzymes or between metabolites.
137
138    Parameters:
139    model (cobra.Model): The COBRA model to be converted into a graph.
140    
141    Returns:
142    nx.DiGraph: A directed bipartite graph representing the model.
143    """
144    G = nx.DiGraph()
145
146    # Add metabolites as nodes
147    for metabolite in model.metabolites:
148        G.add_node(
149            metabolite.id,
150            type="metabolite",
151            name=metabolite.name,
152            formula=metabolite.formula,
153            compartment=metabolite.compartment,
154        )
155
156    for reaction in model.reactions:
157        # Add enzyme node
158        enzyme_node = f"{reaction.id}"
159        G.add_node(enzyme_node, type="enzyme", name=reaction.name)
160        
161        # Add edges from reactants to the reaction
162        for reactant in reaction.reactants:
163            G.add_edge(reactant.id, enzyme_node, type="reactant", reaction_id=reaction.id)
164        
165        # Add edges from the reaction to products
166        for product in reaction.products:
167            G.add_edge(enzyme_node, product.id, type="product", reaction_id=reaction.id)
168        
169        # Create additional edges for reversible reactions
170        if reaction.reversibility:
171            for product in reaction.products:
172                G.add_edge(product.id, enzyme_node, type="reverse_product", reaction_id=reaction.id)
173            for reactant in reaction.reactants:
174                G.add_edge(enzyme_node, reactant.id, type="reverse_reactant", reaction_id=reaction.id)
175
176    return G
